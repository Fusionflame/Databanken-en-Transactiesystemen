update table Divisie(
	add constraint Divisie_Key
		primary key (DivisieID, ClubID)
	add constraint Club_REF 
		foreign key ClubID 
		references Club
		on delete cascade
)

update table Divisie-Team(
	add constraint Divisie-Team_Key
		primary key (DivID, TeamID)
	add constraint DIV_REF 
		foreign key DivID 
		references Divisie
	add constraint TEA_REF 
		foreign key TeamID 
		references Team	
)

update table Team-Speler(
	add constraint Team-Speler_Key
		primary key (TeamID,SpelerID)
	add constraint TEA_REF 
		foreign key TeamID 
		references Team
	add constraint SPE_REF 
		foreign key SpelerID 
		references Speler
)

update table Speler-Fout(
	add constraint Speler-Fout_Key
		primary key (SpelerID,FoutID)
	add constraint SPE_REF 
		foreign key SpelerID 
		references Speler
	add constraint FOU_REF 
		foreign key FoutID 
		references Fout
)

update table Wedstrijd(
	add constraint Wedstrijd_Key
		primary key (CompetitieID, Datum)
	add constraint COM_REF 
		foreign key CompetitieID 
		references Competitie		
)

update table Team-Wedstrijd(
	add constraint Team-Wedstrijd_Key
		primary key (WedstrijdID, TeamID)
	add constraint WED_REF
		foreign key WedstrijdID 
		references Wedstrijd
	add constraint TEA_REF
		foreign key TeamID 
		references Team
)

update table Wedstrijd-Scheidsrechter(
	add constraint Wedstrijd-Scheidsrechter_key
		primary key (WedstrijdID, ScheidsrechterID)
	add constraint WED_REF
		foreign key WedstrijdID 
		references Wedstrijd
	add constraint SCH_REF
		foreign key ScheidsrechterID 
		references Scheidsrechter
)
